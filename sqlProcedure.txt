CREATE PROCEDURE dbo.AddCloth
  @SKUNumber nvarchar(30),
  @Name nvarchar (150),
  @Size nvarchar (5),
  @Color nvarchar (50)
AS 

  BEGIN 
  
    IF EXISTS (SELECT * FROM tbl_cloths WHERE tbl_cloths.SKU = @SKUNumber)
       UPDATE tbl_cloths SET tbl_cloths.Count = tbl_cloths.Count.Count + 1 WHERE tbl_cloths.SKU = @SKUNumber
    ELSE 
    INSERT INTO tbl_cloths (SKU,Name,Size,Color) VALUES (@SKUNumber,@Name,@Size,@Color) 
  END

GO
